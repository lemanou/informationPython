    Function used to check if this is the first time the code runs
    '''
    print "Is this the first time you are running this code? (y/n)"
    yes = set(['yes','y', 'ye', ''])
    no = set(['no','n'])

    choice = raw_input().lower()
    if choice in yes:
       return True
    elif choice in no:
       return False
    else:
       sys.stdout.write("Please respond with 'yes' or 'no'. Exiting...")
       sys.exit()
